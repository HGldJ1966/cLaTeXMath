17 inline static void parseMap(const string& options, _out_ map<string, string>& res) {
18  if (options.empty())
19  return;
20 
21  strtokenizer tokens(options, ",");
22  const int c = tokens.count_tokens();
23  for (int i = 0; i < c; i++) {
24  string tok = tokens.next_token();
25  trim(tok);
26  vector<string> optarg;
27  split(tok, '=', optarg);
28  if (!optarg.empty()) {
29  if (optarg.size() == 2)
30  res[trim(optarg[0])] = trim(optarg[1]);
31  else if (optarg.size() == 1)
32  res[trim(optarg[0])] = "";
33  }
34  }
35 }
